# Global Terrorism Database Collaborative Analysis

# Load necessary libraries
library(dplyr)
library(ggplot2)

# Load the dataset
gtd_data <- read.csv("globalterrorismdb_0718dist.csv", 
                     stringsAsFactors = FALSE)

# Basic data exploration
glimpse(gtd_data)

# ---- Data Wrangling (Yash) ----
gtd_clean <- gtd_data %>%
  filter(!is.na(region_txt), !is.na(nkill))

# ---- Analysis (Yash) ----
# Calculate total fatalities per region
fatalities_by_region <- gtd_clean %>%
  group_by(region_txt) %>%
  summarise(total_fatalities = sum(nkill, na.rm = TRUE)) %>%
  arrange(desc(total_fatalities))

# Print the results
print(fatalities_by_region)

# Visualization (Ravleen)
# Bar chart of total fatalities by region
ggplot(fatalities_by_region, 
  aes(x = reorder(region_txt, -total_fatalities),
      y = total_fatalities)) +
    geom_bar(stat = "identity", fill = "red") +
    labs(title = "Total Fatalities by Region",
        x = "Region",
        y = "Total Fatalities") +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1))
